# Plan 01-04: Navigation Wiring — Full Tab Navigation

## Objective

Wire the full bottom tab navigation so users can navigate between Home (BuzzScreen), History (HistoryScreen), and prepare slots for Settings and Pairing (added in Phase 2). Extract the bottom pill nav from `BuzzScreen` into a shared component (`TabBar.tsx`) so all tab screens share the same nav bar. Update `AppNavigator` to include a `TabNavigator` wrapping the tab screens.

## Current Navigation Structure

```
SplashScreen (custom, not in nav) → fires onFinish
App.tsx: showSplash state → SplashScreen | AppNavigator
AppNavigator: Stack with Login → Home (BuzzScreen)
BuzzScreen: has internal BlurView pill nav (currently no-op for History/Settings)
```

## Target Navigation Structure

```
SplashScreen → fires onFinish
App.tsx: showSplash state → SplashScreen | AppNavigator
AppNavigator:
  Stack:
    - Login (LoginScreen)
    - Tabs (TabNavigator) ← new wrapper
        ├── Home (BuzzScreen)
        ├── History (HistoryScreen)
        └── Settings (placeholder View for now)
```

The tab bar should:

- Use `BlurView` from `expo-blur` for the frosted glass effect
- Show 3 tabs: HOME · HISTORY · SETTINGS

```typescript
import React from 'react';
import { View, Text, TouchableOpacity, StyleSheet } from 'react-native';
import { BlurView } from 'expo-blur';
import { Ionicons } from '@expo/vector-icons';
import { Colors, Typography } from '../theme';

export type TabName = 'Home' | 'History' | 'Settings';

interface TabBarProps {
    activeTab: TabName;
    onNavigate: (tab: TabName) => void;
}

const TABS: Array<{ name: TabName; icon: string; label: string }> = [
    { name: 'Home', icon: 'home', label: 'HOME' },
    { name: 'History', icon: 'time', label: 'HISTORY' },
    { name: 'Settings', icon: 'settings', label: 'SETTINGS' },
];

export function TabBar({ activeTab, onNavigate }: TabBarProps) {
    return (
        <View style={s.wrapper} pointerEvents="box-none">
            <BlurView intensity={80} tint="dark" style={s.pill}>
                <View style={s.tabs}>
                    {TABS.map((tab) => {
                        const isActive = activeTab === tab.name;
                        return (
                            <TouchableOpacity
                                key={tab.name}
                                style={s.tab}
                                onPress={() => onNavigate(tab.name)}
                                activeOpacity={0.7}
                            >
                                <Ionicons
                                    name={tab.icon as any}
                                    size={16}
                                    color={isActive ? Colors.accent : '#555555'}
                                />
                                <Text style={[s.label, isActive && s.labelActive]}>
                                    {tab.label}
                                </Text>
                                {isActive && <View style={s.activeBar} />}
                            </TouchableOpacity>
                        );
                    })}
                </View>
            </BlurView>
        </View>
    );
}

const s = StyleSheet.create({
    wrapper: {
        position: 'absolute',
        bottom: 32,
        left: 0,
        right: 0,
        alignItems: 'center',
        zIndex: 100,
    },
    pill: {
        borderRadius: 999,
        overflow: 'hidden',
        borderWidth: 1,
        borderColor: 'rgba(255,255,255,0.1)',
    },
    tabs: {
        flexDirection: 'row',
        paddingHorizontal: 8,
        paddingVertical: 10,
    },
    tab: {
        alignItems: 'center',
        paddingHorizontal: 20,
        paddingVertical: 4,
        gap: 3,
    },
    label: {
        fontFamily: Typography.pixel,
        fontSize: 9,
        color: '#555555',
        letterSpacing: 1,
    },
    labelActive: {
        color: Colors.accent,
    },
    activeBar: {
        position: 'absolute',
        bottom: -2,
        width: 20,
        height: 2,
        backgroundColor: Colors.accent,
        borderRadius: 1,
    },
});
```

```typescript
import React, { useState } from 'react';
import { View, StyleSheet } from 'react-native';
import { Colors } from '../theme';
import BuzzScreen from '../screens/BuzzScreen';
import HistoryScreen from '../screens/HistoryScreen';
import { TabBar, TabName } from '../components/TabBar';

// Placeholder for Settings — will be replaced in Phase 2
function SettingsPlaceholder() {
    const { Text } = require('react-native');
    const { Typography } = require('../theme');
    return (
        <View style={{ flex: 1, backgroundColor: Colors.background, justifyContent: 'center', alignItems: 'center' }}>
            <Text style={{ fontFamily: Typography.pixel, color: '#555555', fontSize: 13, letterSpacing: 2 }}>
                SETTINGS — COMING SOON
            </Text>
        </View>
    );
}

export function TabNavigator() {
    const [activeTab, setActiveTab] = useState<TabName>('Home');

    const renderScreen = () => {
        switch (activeTab) {
            case 'Home':
                return <BuzzScreen />;
            case 'History':
                return <HistoryScreen />;
            case 'Settings':
                return <SettingsPlaceholder />;
            default:
                return <BuzzScreen />;
        }
    };

    return (
        <View style={s.container}>
            {renderScreen()}
            <TabBar activeTab={activeTab} onNavigate={setActiveTab} />
        </View>
    );
}

const s = StyleSheet.create({
    container: {
        flex: 1,
        backgroundColor: Colors.background,
    },
});
```
